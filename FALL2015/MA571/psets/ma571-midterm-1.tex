\begin{problem}
Let $A\subset X$ and $B\subset Y$. Show that the space $X\times Y$,
\[\clsr{A\times B}=\clsr A\times\clsr B.\]
\end{problem}
\begin{proof}
One containment is clear, namely ``$\supset$'' since
$A\subset\clsr A$ and $B\subset\clsr B$ so $A\times B\subset\clsr
A\times\clsr B$.
\end{proof}
