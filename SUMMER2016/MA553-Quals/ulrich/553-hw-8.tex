\begin{enumerate}[label=(\alph*),noitemsep]

\item $f(X+a)=\sum\Delta_n(f)(a)X^n$, where $a\in\bbk$;
\item $a\in\bbk$ is a root of $f$ of multiplicity $n$ if and only if
  $\Delta_i(f)(a)=0$ for $0\leq i\leq n-1$ and $\Delta_n(f)(a)\neq 0$.
\end{enumerate}
\end{problem}
