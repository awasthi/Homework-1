\begin{problem}[Wheeden \& Zygmund Ch.\@ 7, Ex.\@ 13]
  Prove Theorem 7.36.
\end{problem}
\begin{solution}
  Recall the statement of Theorem 7.36
  \begin{quote}
    \emph{
      \begin{enumerate}[label=\textnormal{(\roman*)}]
      \item If \(\varphi_1\) and \(\varphi_2\) are convex in \((a,b)\),
        then \(\varphi_1+\varphi_2\) is convex in \((a,b)\).
      \item If \(\varphi\) is convex in \((a,b)\) and \(c\) is a positive
        constant, then \(c\varphi\) is convex in \((a,b)\).
      \item If \(\varphi_k\), \(k=1,2,\dotsc\), are convex in \((a,b)\) and
        \(\varphi_k\to\varphi\) in \((a,b)\), then \(\varphi\) is convex in
        \((a,b)\).
      \end{enumerate}
    }
  \end{quote}
\end{solution}
