\begin{problem}(5 pts.)
  Using the method of elimination, determine if the following system of
  equations has a solution, no solution or many solutions:
  \begin{alignat*}{5}
    x_1&+&2x_2&&-3x_3&=&-4\\
    2x_1&+&x_2&&-3x_3&=&4.\nonumber
  \end{alignat*}
\end{problem}

\begin{problem}(5 pts.)
  Given the matrices
  \[
    A\coloneq
    \begin{bmatrix}
      1&-1\\
      -1&1
    \end{bmatrix}
    \qquad
    B\coloneq
    \begin{bmatrix}
      1&1\\
      0&1
    \end{bmatrix}
  \]
  find
  \[
   2A+3B^\rmT.
  \]
\end{problem}
