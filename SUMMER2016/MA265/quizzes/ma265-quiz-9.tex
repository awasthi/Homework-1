\begin{questions}
  \question Let $\calP_2$ be the set of all polynomials of degree less than
  or equal to $2$. We define an inner product on $\calP_2$ by
  \[
    \label{eq:inner-prod}
    \tag{$\bigstar$}
    \langle p(t),q(t) \rangle=\int_{-1}^1 p(t)q(t)\diff t
  \]
  for polynomials $p(t),q(t)\in\calP_2$.
  \begin{parts}
    \part[12] The set $\{1,t,t^2\}$ is a basis for $\calP_2$. Use the
    Gram--Schmidt process to find an orthonormal basis for $\calP_2$ using
    the inner product \eqref{eq:inner-prod}.
    \begin{solution}
      Following the general Gram--Schmidt process, define
      \begin{align*}
        u_1(t)
        &=1\\
        u_2(t)
        &=t%
          -\left[\frac{\langle 1,t \rangle}{\langle 1,1 \rangle}\right]1\\
        &=t-\left[\frac{\int_{-1}^1 t\diff t}{\int_{-1}^1 1\diff t}\right]1\\
        &=t-\left[\frac{1^2-((-1)^2)}{2}\right]1\\
        &=t\\
        u_3(t)
        &=t^2%
          -\left[\frac{\langle t,t^2 \rangle}{\langle t,t \rangle}\right]t%
          -\left[\frac{\langle 1,t \rangle}{\langle 1,1 \rangle}\right]1\\
        &=t^2%
          -\left[\frac{\int_{-1}^1 t^3\diff t}{\int_{-1}^1 t^2\diff t}\right]t%
          -\left[\frac{\int_{-1}^1 t^2\diff t}{\int_{-1}^1 1\diff t}\right]1\\
        &=t^2%
          -\left[\frac{1/4(1)^4-((1/4)(-1)^4)}{2/3}\right]t%
          -\left[\frac{1/3(1)^3-((1/3)(-1)^3)}{2}\right]1\\
        &=t^2-\frac{1}{3}
      \end{align*}
    \end{solution}
    \part[8] Find an orthonormal basis for $\calP_2$. [\emph{Hint:} Use the
    normal basis you found in part (b).]
    \begin{solution}
      Using $\{u_1(t),u_2(t),u_3(t)\}$ we have
      \begin{align*}
        \frac{u_1(t)}{\|u_1(t)\|}&=\frac{1}{\sqrt{\int_{-1}^1 1\diff t}}\\
                                 &=\frac{1}{\sqrt{2}}\\
                                 &=\frac{\sqrt{2}}{2}.\\
        \frac{u_2(t)}{\|u_2(t)\|}&=\frac{t}{\sqrt{\int_{-1}^1 t^2\diff
                                   t}}\\
                                 &=\frac{t}{\sqrt{2/3}}\\
                                 &=\sqrt{\frac{3}{2}}t\\
        \frac{u_3(t)}{\|u_3(t)\|}&=\frac{t^2-1/3}{\sqrt{\int_{-1}^1
                                   (t^2-1/3)(t^2-1/3)\diff t}}\\
                                 &=\sqrt{\frac{45}{8}}\left(t^2-\frac{1}{3}\right)\\
                                 &=\frac{1}{2}\sqrt{\frac{45}{2}}\left(t^2-\frac{1}{3}\right).
      \end{align*}
    \end{solution}
  \end{parts}
\end{questions}
