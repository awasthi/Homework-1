\begin{questions}
  \question Let $V$ and $W$ be subspaces of $\bbR^n$. Suppose $L\colon V\to
  W$ is a surjective map
  %
  \question Does there exist a linear transformation $L\colon V\to W$ is a
  such that
  \[
    \|L(x)\|\leq 1
  \]
  for all $x\in V$?
\end{questions}
