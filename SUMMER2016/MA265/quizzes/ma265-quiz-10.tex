\begin{questions}
  \question

  \question Suppose $L\colon\bbR^n\to\bbR$ is a map that is bounded by $1$,
  i.e., such that for every $\bfx\in\bbR^n$,
  \[
    |L(\bfx)|\leq 1.
  \]
  Can $L$ be a linear map?
\end{questions}
