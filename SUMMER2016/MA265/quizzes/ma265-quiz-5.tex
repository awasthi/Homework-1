\begin{questions}
  \question%
  Consider the subset $S^2\subset\bbR^3$ defined by
  \[
    S^2=\left\{\,(x,y,z)\in\bbR^3:x^2+y^2+z^2-1=0\,\right\}.
  \]
  You may recognize this as the sphere of radius $1$ centered at $(0,0,0)$.
\end{questions}
