\begin{problem}

  value problem
  \[
    \left\{

        \Delta u-u_t&=0&&\text{for \(x\in\Omega\), \(t>0\),}\\
        u(x,0)&=g(x)&&\text{for \(x\in\Omega\),}\\
        u(x,t)&=0&&\text{for \(xi\in\partial\Omega\), \(t\geq 0\),}

    \right.
  \]

  \[
    \left\{
      \begin{aligned}
        \Delta v-v_t&=0&&\text{for \(x\in\bbR^n\), \(t>0\),}\\

      \end{aligned}
