%% Quiz problems
\begin{problem}% [MA 166, Exam 1, Spring 2015]
Let $\bfu=\langle 6,3,1\rangle$, $\bfv=\langle 0,1,2\rangle$, and
$\bfw=\langle 4,-2,5\rangle$.
\begin{enumerate}[label=(\roman*)]
\item Find the scalar projection $\comp_{\bfv}\bfw$.
\item Find the projection $\proj_{\bfu}\bfv$.
\item Find the cross product $\bfv\times\bfw$.
\item What is a vector orthogonal to $\bfv$ and $\bfw$?
\item Find the scalar triple product $\bfu\cdot(\bfv\times\bfw)$.
\item Are the vectors $\bfu$, $\bfv$, and $\bfw$ coplanar?
\end{enumerate}
\end{problem}
\bigskip
\begin{problem}% [MA 166, Exam 1, Fall 2015]
Find the area enclosed by the regions
\begin{enumerate}[label=(\roman*)]
\item $y=x^3$, and $y=|x|$.
\item $y=e^x$, $y=e^2x$, and $x=\ln 2$.
\end{enumerate}
\end{problem}
