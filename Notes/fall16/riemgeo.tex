\section{Differentiable manifolds}
\begin{definition}
  A \emph{differentiable manifold} of dimension \(n\) is a set \(M\) and  a
  family of injective mappings \(\bfx_\alpha\colon U_\alpha\subset\bbR^n\to
  M\)
\end{definition}
